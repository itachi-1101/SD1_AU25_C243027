import java.util.Scanner;

public class P6 {
    public static void main(String[] args) {
        Scanner input=new Scanner(System.in);
        int n=input.nextInt();
        for(int i=1;i<=10;i++)
        {
            long mul=n*i;
            System.out.println(n+" "+"*"+" "+i+" "+"="+" "+mul);
        }
    }
}
