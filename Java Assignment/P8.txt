import java.util.Scanner;

public class P8 {
    public static void main(String[] args) {
        Scanner input=new Scanner(System.in);
        int n=input.nextInt();
        long fac=1;
        for(int i=n;i>0;i--)
        {
            fac*=i;
        }
        System.out.println(fac);
    }
}
